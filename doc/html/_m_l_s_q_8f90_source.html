144 subroutine setmlsqcoefficients(self,particlesXYZIn,panelsXYZIn, particlesVelocityIn,panelsVelocityIn,&

148  real(kreal), intent(in) :: activepanelsvelocityin(:,:), particleshin(:), activepanelshin(:), activepanelsareain(:)
149  !

158  wavg = sum(activepanelsvelocityin(3,:)*activepanelsareain/earth_radius)/(4.0_kreal*earth_radius)
159  havg = sum(activepanelshin(1,:)*activepanelsareain/earth_radius)/(4.0_kreal*earth_radius)

163  datalocs = 0.0_kreal
164  do i=1,self%nNearbyParticles(j)
165  datalocs(:,i) = particlesxyzin(:,self%nearbyParticles(i,j))
166  udata(i) = particlesvelocityin(1,self%nearbyParticles(i,j))
167  vdata(i) = particlesvelocityin(2,self%nearbyParticles(i,j))
168  wdata(i) = particlesvelocityin(3,self%nearbyParticles(i,j))
169  hdata(i) = particleshin(self%nearbyParticles(i,j))
170  enddo
171  do i=1,self%nAdjacentPanels(j)

174  vdata(self%nNearbyParticles(j) + i) = panelsvelocity(2,self%adjacentPanels(i,j))
175  wdata(self%nNearbyParticles(j) + i) = panelsvelocity(3,self%adjacentPanels(i,j))
176  hdata(self%nNearbyParticles(j) + i) = panelsh(self%adjacentPanels(i,j))
177  enddo
178  datalocs(:,self%nNearbyParticles(j) + self%nAdjacentPanels(j) + 1) = panelsxyzin(:,j)
179  udata(self%nNearbyParticles(j) + self%nAdjacentPanels(j) + 1) = panelsvelocity(1,j)
180  vdata(self%nNearbyParticles(j) + self%nAdjacentPanels(j) + 1) = panelsvelocity(2,j)
181  wdata(self%nNearbyParticles(j) + self%nAdjacentPanels(j) + 1) = panelsvelocity(3,j)
182  hdata(self%nNearbyParticles(j) + self%nAdjacentPanels(j) + 1) = panelsh(1,j)

204  a(1:nlocations, 6 ) = datalocs(1,1:npoints) * datalocs(3,1:npoints) ! xz
205  a(1:nlocations, 7 ) = datalocs(2,1:npoints) * datalocs(3,1:npoints) ! yz
206  a(1:nlocations, 8 ) = datalocs(1,1:npoints) * datalocs(1,1:npoints) ! x^2
207  a(1:nlocations, 9 ) = datalocs(2,1:npoints) * datalocs(2,1:npoints) ! y^2
208  a(1:nlocations, 10) = datalocs(3,1:npoints) * datalocs(3,1:npoints) ! z^2
209 
210  at = transpose(a)
211  ata = matmul(at,a)
212 
213  ! find cholesky factorization with lapack
214  call dpotrf('U',10,ata,10,errcode)
215  if ( errcode < 0 ) then

218  elseif ( errcode > 0 ) then

220  endif
221 
222  ! invert ATA with lapack
223  call dpotri('U',10,ata,10,errcode)
224  if ( errcode < 0 ) then

227  elseif (errcode > 0 ) then
228  call logmessage(log,error_logging_level,logkey,'potri ERROR : found zero on diagonal of Cholesky matrix')
229  call logmessage(log,error_logging_level,trim(logkey)//' error found at panel ',j)
230  endif
231 
232  ! Unpack LAPACK'S symmetric storage
233  do m=1,10
234  do k=m+1,10
235  ata(k,m) = ata(m,k)
236  enddo
237  enddo

246  endif
