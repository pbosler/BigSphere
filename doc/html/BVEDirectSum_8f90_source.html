311 subroutine bverk4timestep(self,aMesh,dt,procRank, nProcs)
312 ! Advances aMesh one time step forward using 4th order Runge-Kutta
313 !
314 ! Biot-Savart law is computed in parallel using replicated data on each processor.
315 ! Each process computes its portion of the integral, then broadcasts its results to MPI_WORLD_COMM.
316  type(bverk4data), intent(inout) :: self

318  real(kreal), intent(in) :: dt
319  integer(kint), intent(in) :: procrank, nprocs
320  type(particles), pointer :: aparticles
321  integer(kint) :: j, mpierrcode
322 
323  call logmessage(log,debug_logging_level,logkey,'Entering RK4.')
324 
325  if ( self%rk4IsReady .AND. self%mpiIsReady ) then
326  call zerork4(self)
327  else
328  call logmessage(log,error_logging_level,logkey,'BVERK4 ERROR : data not ready.')
329  return
330  endif
331 
332  aparticles => amesh%particles
333 
334  !!!!!!!!!!!!!!!!!!!!!!!!!!
335  ! RK Stage 1 !
336  !!!!!!!!!!!!!!!!!!!!!!!!!!
337  ! Set input arrays for stage 1
338  self%particlesInput = aparticles%x(:,1:aparticles%N)
339  self%activePanelsInput = self%activePanels%x
340  self%activeVortInput = self%activePanels%relVort
341  self%area = self%activePanels%area
342  self%passivePanelsInput = self%passivePanels%x
343 
344  call bvepassiverhs(self%particlesStage1, self%particlesInput,&
345  self%activePanelsInput, self%activeVortInput,self%area, &
346  self%particlesIndexStart(procrank), self%particlesIndexEnd(procrank))
347  call bvesmoothvelocity(self%passivePanelsStage1, self%passivePanelsInput, &
348  self%activePanelsInput, self%activeVortInput, self%area, &
349  self%passivePanelsIndexStart(procrank), self%passivePanelsIndexEnd(procrank))
350  call bveactiverhs(self%activePanelsStage1, self%activeVortStage1, &
351  self%activePanelsInput, self%activeVortInput, self%area, &
352  self%activePanelsIndexStart(procrank), self%activePanelsIndexEnd(procrank))
353 
354  do j=0,nprocs-1
355  call mpi_bcast(self%particlesStage1(:,self%particlesIndexStart(j):self%particlesIndexEnd(j)), &
356  3*self%particlesMessageSize(j), mpi_double_precision, j, &
357  mpi_comm_world, mpierrcode)
358  call mpi_bcast(self%passivePanelsStage1(:,self%passivePanelsIndexStart(j):self%passivePanelsIndexEnd(j)), &
359  3*self%passivePanelsMessageSize(j), mpi_double_precision, j, &
360  mpi_comm_world, mpierrcode)
361  call mpi_bcast(self%activePanelsStage1(:,self%activePanelsIndexStart(j):self%activePanelsIndexEnd(j)), &
362  3*self%activePanelsMessageSize(j), mpi_double_precision, j, &
363  mpi_comm_world, mpierrcode)
364  call mpi_bcast(self%activeVortStage1(self%activePanelsIndexStart(j):self%activePanelsIndexEnd(j)), &
365  self%activePanelsMessageSize(j), mpi_double_precision, j, &
366  mpi_comm_world, mpierrcode)
367  enddo
368 
369  ! STAGE 1 ONLY : Store velocity and kinetic energy
370  do j=1,aparticles%N
371  aparticles%ke(j) = sum(self%particlesStage1(:,j)*self%particlesStage1(:,j))
372  aparticles%u(:,j) = self%particlesStage1(:,j)
373  enddo
374  do j=1,self%activePanels%N
375  self%activePanels%ke(j) = sum(self%activePanelsStage1(:,j)*self%activePanelsStage1(:,j))
376  self%activePanels%u(:,j) = self%activePanelsStage1(:,j)
377  enddo
378  amesh%totalE = 0.5*sum(self%activePanels%ke*self%area)
379 
380 
381  self%particlesStage1 = dt*self%particlesStage1
382  self%passivePanelsStage1 = dt*self%passivePanelsStage1
383  self%activePanelsStage1 = dt*self%activePanelsStage1
384  self%activeVortStage1 = dt*self%activeVortStage1
385 
386  !!!!!!!!!!!!!!!!!!!!!!!!!!
387  ! RK Stage 2 !
388  !!!!!!!!!!!!!!!!!!!!!!!!!!
389  ! Set input arrays for stage 2
390  self%particlesInput = aparticles%x(:,1:aparticles%N) + 0.5_kreal*self%particlesStage1
391  self%activePanelsInput = self%activePanels%x + 0.5_kreal*self%activePanelsStage1
392  self%activeVortInput = self%activePanels%relVort + 0.5_kreal*self%activeVortStage1
393  self%area = self%activePanels%area
394  self%passivePanelsInput = self%passivePanels%x + 0.5_kreal*self%passivePanelsStage1
395 
396  call bvepassiverhs(self%particlesStage2, self%particlesInput,&
397  self%activePanelsInput, self%activeVortInput,self%area, &
398  self%particlesIndexStart(procrank), self%particlesIndexEnd(procrank))
399  call bvesmoothvelocity(self%passivePanelsStage2, self%passivePanelsInput, &
400  self%activePanelsInput, self%activeVortInput, self%area, &
401  self%passivePanelsIndexStart(procrank), self%passivePanelsIndexEnd(procrank))
402  call bveactiverhs(self%activePanelsStage2, self%activeVortStage2, &
403  self%activePanelsInput, self%activeVortInput, self%area, &
404  self%activePanelsIndexStart(procrank), self%activePanelsIndexEnd(procrank))
405

407  call mpi_bcast(self%particlesStage2(:,self%particlesIndexStart(j):self%particlesIndexEnd(j)), &
408  3*self%particlesMessageSize(j), mpi_double_precision, j, &
409  mpi_comm_world, mpierrcode)

411  3*self%passivePanelsMessageSize(j), mpi_double_precision, j, &

415  mpi_comm_world, mpierrcode)

434  self%passivePanelsInput = self%passivePanels%x + 0.5_kreal*self%passivePanelsStage2

615 subroutine bveactiverhs(dX, dVort, pointVorts, vort, area, indexStart, indexEnd)
616 ! Computes the Biot-Savart law for active particles.
617 ! Skips singular panel.
618 !
619  real(kreal), intent(out) :: dx(:,:)
620  real(kreal), intent(out) :: dvort(:)
621  real(kreal), intent(in) :: pointvorts(:,:)
622  real(kreal), intent(in) :: vort(:)
623  real(kreal), intent(in) :: area(:)
624  integer(kint), intent(in) :: indexstart, indexend
625  integer(kint) :: nn, j, k
626  real(kreal) :: denom
627 
628  ! Error checking
629  nn = size(dx,2)
630  if ( nn /= size(dvort) .OR. nn /= size(pointvorts,2) ) then
631  call logmessage(log,error_logging_level,'BVEActiveRHS ERROR : ','size mismatch 1.')
632  return
633  endif
634  if ( nn /= size(vort) .OR. nn /= size(area) ) then
635  call logmessage(log,error_logging_level,'BVEActiveRHS ERROR : ','size mismatch 2.')
636  return
637  endif
638 
639  dx = 0.0_kreal
640  dvort = 0.0_kreal
641  do j=indexstart, indexend
642  do k=1,j-1
643  denom = earth_radius*earth_radius - sum(pointvorts(:,k)*pointvorts(:,j))
644  dx(1,j) = dx(1,j) + (pointvorts(2,j)*pointvorts(3,k) - pointvorts(3,j)*pointvorts(2,k))*vort(k)*area(k)/denom
645  dx(2,j) = dx(2,j) + (pointvorts(3,j)*pointvorts(1,k) - pointvorts(1,j)*pointvorts(3,k))*vort(k)*area(k)/denom
646  dx(3,j) = dx(3,j) + (pointvorts(1,j)*pointvorts(2,k) - pointvorts(2,j)*pointvorts(1,k))*vort(k)*area(k)/denom
647  enddo
648  do k=j+1,nn
649  denom = earth_radius*earth_radius - sum(pointvorts(:,k)*pointvorts(:,j))
650  dx(1,j) = dx(1,j) + (pointvorts(2,j)*pointvorts(3,k) - pointvorts(3,j)*pointvorts(2,k))*vort(k)*area(k)/denom
651  dx(2,j) = dx(2,j) + (pointvorts(3,j)*pointvorts(1,k) - pointvorts(1,j)*pointvorts(3,k))*vort(k)*area(k)/denom
652  dx(3,j) = dx(3,j) + (pointvorts(1,j)*pointvorts(2,k) - pointvorts(2,j)*pointvorts(1,k))*vort(k)*area(k)/denom
653  enddo
654  enddo

656  dvort = -2.0_kreal*omega*dx(3,:)/earth_radius
