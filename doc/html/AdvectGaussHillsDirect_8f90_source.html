320  real(kreal) :: ghillsexact
321  real(kreal), intent(in) :: xyz(3), hmax, beta
322  !
323  real(kreal) :: xc1(3), xc2(3), h1, h2
324 
325  xc1 = [ cos(5.0_kreal * pi / 6.0_kreal), sin( 5.0_kreal * pi / 6.0_kreal ), 0.0_kreal ]
