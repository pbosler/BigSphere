847 function quadraticscalarappx( coeffs, xyz)
848  real(kreal) :: quadraticscalarappx
849  real(kreal), intent(in) :: coeffs(10), xyz(3)
850  quadraticscalarappx = coeffs(1) + coeffs(2) * xyz(1) + coeffs(3) * xyz(2) + coeffs(4) * xyz(3) + &
851  coeffs(5) * xyz(1) * xyz(2) + coeffs(6) * xyz(1) * xyz(3) + coeffs(7) * xyz(2) * xyz(3) + &
852  coeffs(8) * xyz(1) * xyz(1) + coeffs(9) * xyz(2) * xyz(2) + coeffs(10)* xyz(3) * xyz(3)
853 end function
854 
855 function quadraticgradappx( coeffs, xyz)
856  real(kreal) :: quadraticgradappx(3)
857  real(kreal), intent(in) :: coeffs(10), xyz(3)
858  !
859  real(kreal) :: cartgrad(3), dotprod
860 
861  cartgrad = [ coeffs(2) + coeffs(5) * xyz(2) + coeffs(6) * xyz(3) + 2.0_kreal * coeffs(8) * xyz(1) , &
862  coeffs(3) + coeffs(5) * xyz(1) + coeffs(7) * xyz(3) + 2.0_kreal * coeffs(9) * xyz(2) , &
863  coeffs(4) + coeffs(6) * xyz(1) + coeffs(7) * xyz(2) + 2.0_kreal * coeffs(10) * xyz(3) ]
864 
865  dotprod = sum( xyz * cartgrad)
866 
867  quadraticgradappx = cartgrad - dotprod * xyz / earth_radius / earth_radius
868 end function
869 
870 function quadraticlaplacianappx( coeffs, xyz )
871  real(kreal) :: quadraticlaplacianappx
872  real(kreal), intent(in) :: coeffs(10), xyz(3)
873  !
874  real(kreal) :: projterm, sqrad, term_x, term_y, term_z
875 
876  sqrad = earth_radius * earth_radius
877 
878  projterm = coeffs(2) * xyz(1) + coeffs(3) * xyz(2) + coeffs(4) * xyz(3) + 2.0_kreal * coeffs(5) * xyz(1) * xyz(2) + &
879  2.0_kreal * coeffs(6) * xyz(1) * xyz(3) + 2.0_kreal * coeffs(7) * xyz(2) * xyz(3) + &
880  2.0_kreal * coeffs(8) * xyz(1) * xyz(1) + 2.0_kreal * coeffs(9) * xyz(2) * xyz(2) + &
881  2.0_kreal * coeffs(10)* xyz(3) * xyz(3)
