162  procedure(settraceronmesh) :: updatetraceronmesh
163  type(tracersetup), intent(in) :: tracerdef
164  type(refinementsetup), intent(in) :: refinerelvort

166  type(bvesetup), intent(in) :: vortdef

168  integer(kint) :: refinecount, spaceleft, j, counter1, counter2

174  ! check for invalid states
175  if ( refinetracer%type == tracer_refine .AND. ( getntracer(amesh%panels) < refinetracer%tracerID ) ) then
176  call logmessage(log,error_logging_level,'InitialRefinement ERROR : ','invalid tracer number.')
177  return
178  endif
179  if ( refinetracer%type /= null_refine .AND. refinetracer%type /= tracer_refine ) then
180  call logmessage(log,error_logging_level,'InitialRefinement ERROR : ','invalid tracer refinement type.')
181  return
182  endif
183  if ( refinerelvort%type /= null_refine .AND. refinerelvort%type /= relvort_refine ) then
184  call logmessage(log,error_logging_level,'InitialRefinement ERROR : ','invalid relVort refinement type.')
185  return
186  endif
187  if ( refinerelvort%type == null_refine .AND. refinetracer%type == null_refine ) then
188  call logmessage(log,warning_logging_level,'InitialRefinement WARNING : ','NULL refinement data.')
189  return
190  endif
191 
192  apanels => amesh%panels
193  allocate(refineflag(apanels%N_Max))
194  refineflag = .false.
195  keepgoing = .false.
196 
197  !
198  ! Apply refinement criteria
199  !
200  limit = 0
201  startindex = 1
202  if ( refinetracer%type /= null_refine ) then
203  limit = max(limit,refinetracer%limit)
204  call flagpanelsfortracermaxrefinement(refineflag,amesh,refinetracer,startindex)
205  counter1 = count(refineflag)
206  call flagpanelsfortracervariationrefinement(refineflag,amesh,refinetracer,startindex)
207  counter2 = count(refineflag) - counter1
208  write(formatstring,'(A)') '(A,I8,A)'
209  write(logstring,formatstring) 'tracerMax criterion triggered ', counter1, ' times.'
210  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
211  write(logstring,formatstring) 'tracerVar criterion triggered ', counter2, ' times.'
212  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
213  endif
214  if ( refinerelvort%type /= null_refine) then
215  limit = max(limit,refinerelvort%limit)
216  counter1 = count(refineflag)
217  call flagpanelsforcirculationrefinement(refineflag,amesh,refinerelvort,startindex)
218  counter1 = count(refineflag) - counter1
219  call flagpanelsforrelvortvariationrefinement(refineflag,amesh,refinerelvort,startindex)
220  counter2 = count(refineflag) - counter1
221  write(formatstring,'(A)') '(A,I8,A)'
222  write(logstring,formatstring) 'circMax criterion triggered ', counter1, ' times.'
223  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
224  write(logstring,formatstring) 'relVortVar criterion triggered ', counter2, ' times.'
225  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
226  endif
227

229  spaceleft = apanels%N_Max - apanels%N
230

232  ! exit if refinement is not needed
233  !

236  deallocate(refineflag)
237  return
238  endif
239 
240  !
241  ! check for memory, exit if insufficient
242  !
243  if ( spaceleft/4 < refinecount ) then
244  call logmessage(log,warning_logging_level,'InitRefine : WARNING ',' insufficient memory for AMR.')
245  deallocate(refineflag)
246  return
247  endif
248 
249  keepgoing = .true.
250  amrloopcounter = 0
251 
252  do while (keepgoing)
253  amrloopcounter = amrloopcounter + 1
254  write(logstring,'(A,I2,A,I8,A)') 'AMR loop ',amrloopcounter,' : refining ',refinecount,' panels.'
255  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
256  !
257  ! Divide flagged panels
258  !
259  noldpanels = apanels%N
260  do j=startindex, apanels%N
261  if ( refineflag(j) ) then
262  call dividepanel(amesh,j)
263  refineflag(j) = .false.
264  endif
265  enddo
266  !
267  ! Ensure adjacent panels differ by no more than one mesh level
268  !
269  call flagpanelsatrefinementboundaries(refineflag,amesh)
270  do j=1,apanels%N
271  if ( refineflag(j) ) then
272  call dividepanel(amesh,j)
273  refineflag(j) = .false.
274  endif
275  enddo
276  !
277  ! Set data on refined mesh
278  !
279  if ( amesh%nTracer > 0 ) call updatetraceronmesh(amesh,tracerdef)
280  if ( amesh%problemKind == bve_solver ) call updatevorticityonmesh(amesh,vortdef)
281  !
282  ! Prevent too much refinement
283  !
284  if ( amrloopcounter >= limit ) then
285  keepgoing = .false.
286  call logmessage(log,warning_logging_level,'InitRefine WARNING : ',' refinement limit reached.')
287  endif
288  !
289  ! Apply refinement criteria
290  !
291  startindex = noldpanels+1
292  noldpanels = apanels%N
293  if ( refinetracer%type /= null_refine ) then
294  call flagpanelsfortracermaxrefinement(refineflag,amesh,refinetracer,startindex)
295  counter1 = count(refineflag)
296  call flagpanelsfortracervariationrefinement(refineflag,amesh,refinetracer,startindex)
297  counter2 = count(refineflag) - counter1
298  write(formatstring,'(A)') '(A,I8,A)'
299  write(logstring,formatstring) 'tracerMax criterion triggered ', counter1, ' times.'
300  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
301  write(logstring,formatstring) 'tracerVar criterion triggered ', counter2, ' times.'
302  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
303  endif
304  if ( refinerelvort%type /= null_refine) then
305  counter1 = count(refineflag)
306  call flagpanelsforcirculationrefinement(refineflag,amesh,refinerelvort,startindex)
307  counter1 = count(refineflag) - counter1
308  counter2 = count(refineflag)
309  call flagpanelsforrelvortvariationrefinement(refineflag,amesh,refinerelvort,startindex)
310  counter2 = count(refineflag) - counter2
311  write(formatstring,'(A)') '(A,I8,A)'
312  write(logstring,formatstring) 'circMax criterion triggered ', counter1, ' times.'
313  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
314  write(logstring,formatstring) 'relVortVar criterion triggered ', counter2, ' times.'
315  call logmessage(log,trace_logging_level,'InitRefine : ',logstring)
316  endif
317 
318  refinecount = count(refineflag)
319 
320  !
321  ! exit if refinement is not needed
322  !
323  if ( refinecount == 0 ) then
324  call logmessage(log,trace_logging_level,'InitRefine : ','refinement converged.')
325  keepgoing = .false.
326  endif
327 
328  !
329  ! check for memory, exit if insufficient
330  !
331  if ( spaceleft/4 < refinecount ) then
332  call logmessage(log,warning_logging_level,'InitRefine : ',' insufficient memory for AMR.')
333  keepgoing = .false.
334  endif
335  enddo
336  deallocate(refineflag)
337 end subroutine

339 subroutine lagrangianremeshprivate(aMesh, setVorticity, vortDef, vortRefine, &
340  settracer, tracerdef, tracerrefine, &
341  flowmaprefine, interpsmoothtol)
342 ! Performs a Lagrangian remeshing of a SphereMesh object.
343 ! The Lagrangian paramater is interpolated from the old mesh to a new mesh.
344 ! Materially invariant data (tracers or vorticity) are assigned to the new mesh via the interfaces "setVorticity"
345 ! and "setTracer," whose inputs are the interpolated Lagrangian parameters.
346 ! SSRFPACK provides the interpolation scheme (cubic Hermite polynomials on the Delaunay triangulation of SphereMesh particles).
347 !
348 ! If AMR is in use, this subroutine adaptively refines the new mesh using the input criteria as well.
349 !
350 ! NewMesh data are copied into the old mesh object, overwriting the previous data.
351 !
352  ! calling parameters
353  type(spheremesh), intent(inout) :: amesh
354  procedure(setvorticityonmesh) :: setvorticity
355  type(bvesetup), intent(in) :: vortdef
356  type(refinementsetup), intent(in) :: vortrefine
357  procedure(settraceronmesh) :: settracer
358  type(tracersetup), intent(in) :: tracerdef
359  type(refinementsetup), intent(in) :: tracerrefine
360  type(refinementsetup), intent(in) :: flowmaprefine
361  real(kreal), intent(in), optional :: interpsmoothtol
362  ! local variables
363  type(stripackdata) :: deltri
364  type(ssrfpackdata) :: lagsource
365  logical(klog) :: vectorinterp
366  type(spheremesh) :: newmesh
367  type(particles), pointer :: newparticles
368  type(edges), pointer :: newedges
369  type(panels), pointer :: newpanels
370  integer(kint) :: j, amrloopcounter, counter1, counter2
371  logical(klog), allocatable :: refineflag(:)
372  logical(klog) :: keepgoing, refinetracer, refinevort, refineflowmap
373  integer(kint) :: startindex, noldpanels, noldparticles, refinecount, spaceleft, limit
374 
375 
376  nullify(newparticles)
377  nullify(newedges)
378  nullify(newpanels)
379  vectorinterp = .true.
380  refineflowmap = .false.
381  refinetracer = .false.
382  refinevort = .false.
383 
384  call logmessage(log,debug_logging_level,logkey,' entering Lagrangian remesh.')
385 
386  !
387  ! determine what types of AMR to use
388  !
389  if ( tracerrefine%type == tracer_refine .AND. &
390  getntracer(amesh%panels) <= tracerrefine%tracerID ) refinetracer = .true.
391  if ( vortrefine%type == relvort_refine ) refinevort = .true.
392  if ( flowmaprefine%type == flowmap_refine) refineflowmap = .true.
393 
394  !
395  ! set existing mesh as data source for interpolation
396  !
397  call new(deltri,amesh)
398  call delaunaytriangulation(deltri)
399  call new(lagsource,deltri,vectorinterp)
400  if ( present(interpsmoothtol) ) call setsigmatol(lagsource,interpsmoothtol)
401  call setsourcelagrangianparameter(lagsource,deltri)
402 
403  call logmessage(log,debug_logging_level,logkey,' remesh source data ready.')
404 
405  !
406  ! Build a new mesh
407  !
408  call new(newmesh,amesh%panelKind,amesh%initNest,amesh%AMR,amesh%nTracer,amesh%problemKind)
409  newparticles => newmesh%particles
410  newedges => newmesh%edges
411  newpanels => newmesh%panels
412 
413  !
414  ! interpolate lagrangian parameter from old mesh to new mesh
415  !
416  do j=1,newparticles%N
417  newparticles%x0(:,j) = interpolatevector(newparticles%x(:,j),lagsource,deltri)
418  ! renormalize to spherical surface
419  newparticles%x0(:,j) = newparticles%x0(:,j) / &
420  sqrt(sum(newparticles%x0(:,j)*newparticles%x0(:,j)))*earth_radius
421  enddo
422  do j=1,newpanels%N
423  newpanels%x0(:,j) = interpolatevector(newpanels%x(:,j),lagsource,deltri)
424  ! renormalize
425  newpanels%x0(:,j) = newpanels%x0(:,j) / &
426  sqrt(sum(newpanels%x0(:,j)*newpanels%x0(:,j)))*earth_radius
427  enddo
428  !
429  ! set tracer values on new mesh
430  !
431  if ( amesh%nTracer > 0 ) call settracer(newmesh,tracerdef)
432  !
433  ! set vorticity values on new mesh
434  !
435  if ( amesh%problemKind == bve_solver ) call setvorticity(newmesh,vortdef)
436 
437  call logmessage(log,debug_logging_level,logkey,' new uniform mesh ready.')
438 
439  !
440  ! AMR
441  !
442  if ( amesh%AMR > 0 ) then
443  allocate(refineflag(newpanels%N_Max))
444  refineflag = .false.
445  startindex = 1
446  keepgoing = .false.
447  limit = 0
448 
449  !
450  ! Apply refinement criteria
451  !
452  if ( refinetracer ) then
453  limit = max(limit,tracerrefine%limit)
454  call flagpanelsfortracermaxrefinement(refineflag,newmesh,tracerrefine,startindex)
455  counter1 = count(refineflag)
456  call flagpanelsfortracervariationrefinement(refineflag,newmesh,tracerrefine,startindex)
457  counter2 = count(refineflag) - counter1
458  write(formatstring,'(A)') '(A,I8,A)'
459  write(logstring,formatstring) 'LagRemesh AMR tracerMax criterion triggered ', counter1, ' times.'
460  call logmessage(log,trace_logging_level,logkey,logstring)
461  write(logstring,formatstring) 'LagRemesh AMR tracerVar criterion triggered ', counter2, ' times.'
462  call logmessage(log,trace_logging_level,logkey,logstring)
463  endif
464  if ( refinevort) then
465  limit = max(limit,vortrefine%limit)
466  counter1 = count(refineflag)
467  call flagpanelsforcirculationrefinement(refineflag,newmesh,vortrefine,startindex)
468  counter1 = count(refineflag) - counter1
469  call flagpanelsforrelvortvariationrefinement(refineflag,newmesh,vortrefine,startindex)
470  counter2 = count(refineflag) - counter1
471  write(formatstring,'(A)') '(A,I8,A)'
472  write(logstring,formatstring) 'LagRemesh AMR circMax criterion triggered ', counter1, ' times.'
473  call logmessage(log,trace_logging_level,logkey,logstring)
474  write(logstring,formatstring) 'LagRemesh AMR relVortVar criterion triggered ', counter2, ' times.'
475  call logmessage(log,trace_logging_level,logkey,logstring)
476  endif
477  if ( refineflowmap ) then
478  limit = max(limit,flowmaprefine%limit)
479  counter1 = count(refineflag)
480  call flagpanelsforflowmaprefinement(refineflag,newmesh,flowmaprefine,startindex)
481  counter1 = count(refineflag) - counter1
482  write(formatstring,'(A)') '(A,I8,A)'
483  write(logstring,formatstring) 'LagRemesh AMR flowMap variation criterion triggered ', counter1, ' times.'
484  call logmessage(log,trace_logging_level,logkey,logstring)
485  endif
486 
487  refinecount = count(refineflag)
488  spaceleft = newpanels%N_Max - newpanels%N
489 
490  !
491  ! exit if refinement is not needed, or insufficient memory
492  !
493  if ( refinecount == 0 ) then
494  call logmessage(log,trace_logging_level,logkey,'LagRemesh : no refinement necessary.')
495  keepgoing = .false.
496  elseif ( spaceleft/4 < refinecount ) then
497  call logmessage(log,warning_logging_level,logkey,'LagRemesh : insufficient memory for AMR.')
498  keepgoing = .false.
499  else
500  keepgoing = .true.
501  endif
502 
503  amrloopcounter = 0
504 
505  do while (keepgoing)
506  amrloopcounter = amrloopcounter + 1
507 
508  write(logstring,'(A,I3,A,I8,A)') 'LagRemesh loop ',amrloopcounter,' : refining ',refinecount,' panels.'
509  call logmessage(log,trace_logging_level,logkey,logstring)
510  !
511  ! divide flagged panels
512  !
513  noldpanels = newpanels%N
514  noldparticles = newparticles%N
515  do j=startindex,newpanels%N
516  if ( refineflag(j) ) then
517  call dividepanel(newmesh,j)
518  refineflag(j) = .false.
519  endif
520  enddo
521  !
522  ! ensure adjacent panels differ by no more than one level
523  !
524  call flagpanelsatrefinementboundaries(refineflag,newmesh)
525  do j=1,newpanels%N
526  if ( refineflag(j) ) then
527  call dividepanel(newmesh,j)
528  refineflag(j) = .false.
529  endif
530  enddo
531 
532  !
533  ! set problem data on mesh
534  !
535  do j=noldparticles+1,newparticles%N
536  newparticles%x0(:,j) = interpolatevector(newparticles%x(:,j),lagsource,deltri)
537  newparticles%x0(:,j) = newparticles%x0(:,j) / &
538  sqrt(sum(newparticles%x0(:,j)*newparticles%x0(:,j)))*earth_radius
539  enddo
540  do j=noldpanels+1,newpanels%N
541  newpanels%x0(:,j) = interpolatevector(newpanels%x(:,j),lagsource,deltri)
542  newpanels%x0(:,j) = newpanels%x0(:,j) / &
543  sqrt(sum(newpanels%x0(:,j)*newpanels%x0(:,j)))*earth_radius
544  enddo
545  if ( amesh%nTracer > 0 ) call settracer(newmesh,tracerdef)
546  if ( amesh%problemKind == bve_solver) call setvorticity(newmesh,vortdef)
547 
548  !
549  ! prevent too much refinement
550  !
551  if ( amrloopcounter >= limit ) then
552  keepgoing = .false.
553  call logmessage(log,warning_logging_level,logkey,'LagRemesh WARNING : refinement limit reached.')
554  endif
555

557  ! apply refinement criteria
558  !
559  startindex = noldpanels + 1
560  noldpanels = newpanels%N
561  if ( refinetracer ) then
562  limit = max(limit,tracerrefine%limit)
563  call flagpanelsfortracermaxrefinement(refineflag,newmesh,tracerrefine,startindex)
564  counter1 = count(refineflag)
565  call flagpanelsfortracervariationrefinement(refineflag,newmesh,tracerrefine,startindex)
566  counter2 = count(refineflag) - counter1
567  write(formatstring,'(A)') '(A,I8,A)'
568  write(logstring,formatstring) 'LagRemesh AMR :tracerMax criterion triggered ', counter1, ' times.'
569  call logmessage(log,trace_logging_level,logkey,logstring)
570  write(logstring,formatstring) 'LagRemesh AMR : tracerVar criterion triggered ', counter2, ' times.'
571  call logmessage(log,trace_logging_level,logkey,logstring)
572  endif
573  if ( refinevort) then
574  limit = max(limit,vortrefine%limit)
575  counter1 = count(refineflag)
576  call flagpanelsforcirculationrefinement(refineflag,newmesh,vortrefine,startindex)
577  counter1 = count(refineflag) - counter1
578  call flagpanelsforrelvortvariationrefinement(refineflag,newmesh,vortrefine,startindex)
579  counter2 = count(refineflag) - counter1
580  write(formatstring,'(A)') '(A,I8,A)'
